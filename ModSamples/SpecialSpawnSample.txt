--SpecialSpawns are applied to the zone to make a change to the terrain or spawn items.
--Apply them to the zone using the "specialSpawn" TriggerEffect.

--Here are some example SpecialSpawns:

--50% chance to change terrain to dirt wherever there is Iron Armor
[SpecialSpawn] ID=mySpecialSpawn_dirtUnderArmor;
offsetsToCheck=0,0; 
IDsToCheckFor=armor_iron; 

terrainID=dirt; 
percentChance=0.5;

--90% chance to spawn Iron Armor anywhere with two tiles of dirt to the North
[SpecialSpawn] ID=mySpecialSpawn_ArmorSouthOfDirts;
offsetsToCheck=0,-1; 
IDsToCheckFor=dirt; 
offsetsToCheck=0,-2; 
IDsToCheckFor=dirt; 

spawnID=armor_iron; 
percentChance=0.9;













--here is a list of all the possible values you can use in SpecialSpawn along with definitions.
[SpecialSpawn] ID=specialSpawnListOfAllValues;
--tile offsets to check
offsetsToCheck=0,0; 

--itemTypeID, terrainTypeID to check for.
--Can include multiple values by separating with |
--can also use syntax "!thing" for not thing
--also includes special values "noItems", "noActors", and "zoneEdge"
--example: IDsToCheckFor=zoneEdge|armor_iron|!dirt; 
IDsToCheckFor=; 

--if IDsToCheckFor is empty or not found, check for terrainElement
elementToCheckFor=earth;

--if IDsToCheckFor and elementToCheckFor are empty or not found, check for itemTypeSpecialProperty
itemTypeSpecialPropertyToCheckFor=pushable;

--itemTypeID, SpawnData, or ActorPrefab to spawn (if any)
spawnID=;
--terrainTypeID to change tile to (if any)
terrainID=;

--amount of items to spawn (if applicable)
minQuantity = 1;
maxQuantity = 1;

minDistFromZoneBounds = 0;
minDistFromPlayer = 0;

--negative numbers make low values more likely, and vice versa.
quantityInfluenceMod = 0;

--sets rot if applyRot is not -1
applyRot = -1;

--from 0 to 1
percentChance = 1;

--remove existing items from the tile if there's a match
removeItems = false;

--only spawns the item if it can fit
itemMustFit = false;

--if true, doesn't check every tile; instead, grabs random tiles percentChance*zoneSize times
quickSpawnMode = false;

--if true, uses standard procedure for changing terrain & modifies adjacent tiles to fit wallType/wallBottomType (only for use with levelData)
adjustWallTypesAccordingly = true;
